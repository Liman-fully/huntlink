**1. CacheService（cache.service.ts）**
```typescript
// backend/src/common/cache/cache.service.ts
import { Injectable, Inject } from '@nestjs/common';
import { Redis } from 'ioredis';

@Injectable()
export class CacheService {
  private readonly DEFAULT_TTL = 300; // 5 分钟

  constructor(@Inject('REDIS_CLIENT') private readonly redis: Redis) {}

  async get<T>(key: string): Promise<T | null> {
    const data = await this.redis.get(key);
    if (!data) return null;
    try {
      return JSON.parse(data) as T;
    } catch {
      return null;
    }
  }

  async set(key: string, value: any, ttl: number = this.DEFAULT_TTL): Promise<void> {
    await this.redis.setex(key, ttl, JSON.stringify(value));
  }

  async del(key: string): Promise<void> {
    await this.redis.del(key);
  }

  async keys(pattern: string): Promise<string[]> {
    return await this.redis.keys(pattern);
  }

  // 生成缓存键
  generateCacheKey(prefix: string, params: Record<string, any>): string {
    const sortedParams = Object.keys(params)
      .sort()
      .map((key) => `${key}=${params[key]}`)
      .join('&');
    return `${prefix}:${sortedParams}`;
  }
}
```

**4. 缓存命中率监控**
```typescript
// backend/src/common/cache/cache.metrics.ts
@Injectable()
export class CacheMetrics {
  private hits = 0;
  private misses = 0;

  recordHit(): void {
    this.hits++;
  }

  recordMiss(): void {
    this.misses++;
  }

  getStats(): CacheStats {
    const total = this.hits + this.misses;
    return {
      hits: this.hits,
      misses: this.misses,
      hitRate: total > 0 ? (this.hits / total) * 100 : 0,
    };
  }
}
```

```typescript
// backend/src/modules/export/export.service.ts
@Injectable()
export class ExportService {
  constructor(
    @InjectRepository(Candidate)
    private candidateRepo: Repository<Candidate>,
    private queue: BullQueue,
    private storageService: StorageService,
  ) {}

  // 创建导出任务
  async createExportTask(
    userId: string,
    candidateIds: string[],
    format: 'pdf' | 'excel',
  ): Promise<ExportTask> {
    const task = await this.exportTaskRepo.save({
      userId,
      candidateIds,
      format,
      status: 'pending',
      createdAt: new Date(),
    });

    // 加入队列
    await this.queue.add('export', {
      taskId: task.id,
      candidateIds,
      format,
    });

    return task;
  }

  // 执行导出
  async processExport(taskId: string, candidateIds: string[], format: string): Promise<void> {
    // 更新状态为 processing
    await this.exportTaskRepo.update(taskId, { status: 'processing' });

    // 查询候选人数据
    const candidates = await this.candidateRepo.findByIds(candidateIds);

    // 生成文件
    let filePath: string;
    if (format === 'pdf') {
      filePath = await this.generatePdf(candidates);
    } else {
      filePath = await this.generateExcel(candidates);
    }

    // 上传到 COS
    const url = await this.storageService.upload(filePath);

    // 更新任务状态
    await this.exportTaskRepo.update(taskId, {
      status: 'completed',
      fileUrl: url,
      completedAt: new Date(),
    });
  }

  // 查询导出任务状态
  async getTaskStatus(taskId: string): Promise<ExportTask> {
    return await this.exportTaskRepo.findOne(taskId);
  }
}
```

```typescript
    const stream = fs.createWriteStream(filePath);

    doc.moveDown();

    // 添加每个候选人的信息

      doc.fontSize(10).text(`电话：${candidate.phone}`);
      doc.text(`邮箱：${candidate.email}`);
      doc.text(`经验：${candidate.experience}`);
      doc.text(`技能：${candidate.skills.join(', ')}`);
      doc.moveDown();
    }

    doc.end();

    return new Promise((resolve) => {
      stream.on('finish', () => resolve(filePath));
    });
  }
}
```

```typescript
import ExcelJS from 'exceljs';

export class ExcelExporter {
  async generate(candidates: Candidate[]): Promise<string> {
    const workbook = new ExcelJS.Workbook();
    const worksheet = workbook.addWorksheet('候选人列表');

    // 添加表头
    worksheet.columns = [
      { header: '姓名', key: 'name', width: 20 },
      { header: '电话', key: 'phone', width: 15 },
      { header: '邮箱', key: 'email', width: 30 },
      { header: '经验', key: 'experience', width: 15 },
      { header: '学历', key: 'education', width: 15 },
      { header: '技能', key: 'skills', width: 40 },
    ];

    // 添加数据
    candidates.forEach((candidate) => {
      worksheet.addRow({
        name: candidate.name,
        phone: candidate.phone,
        email: candidate.email,
        experience: candidate.experience,
        education: candidate.education,
        skills: candidate.skills.join(', '),
      });
    });

    const filePath = `/tmp/export-${Date.now()}.xlsx`;
    await workbook.xlsx.writeFile(filePath);

    return filePath;
  }
}
```
